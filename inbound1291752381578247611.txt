package buylaptop;

public class diagram {


	Customer customer;
	Laptop laptop;
	public String date ;
	public int qty;
	
	public BuyLaptop(Customer customer,Laptop laptop,String date,int qty)
	{
		this.customer=customer;
		this.laptop=laptop;
		this.date=date;
		this.qty=qty;
